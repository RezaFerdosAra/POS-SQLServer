CREATE   PROC  [operation].[SP_SoftDeleteCustomer]  @CustomerID INT
WITH ENCRYPTION
AS
BEGIN
  SET NOCOUNT ON
  BEGIN TRY
    UPDATE [identity].Customers
    SET IsActive = 0
    WHERE CustomerID = @CustomerID
  END TRY
  BEGIN CATCH
    RAISERROR(N'خطا در حذف مشتری', 16, 1)
    ROLLBACK TRAN
  END CATCH
END
GO
