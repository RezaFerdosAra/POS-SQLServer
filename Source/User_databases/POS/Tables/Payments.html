CREATE TRIGGER [finance].[tr_UpdateAmount] ON [finance].[Payments]
AFTER INSERT
AS
BEGIN
   SET NOCOUNT ON

 UPDATE Payments
 SET Amount = i.TotalAmount
 FROM Invoices i JOIN (SELECT * FROM INSERTED) ins ON ins.InvoiceID = i.InvoiceID
 WHERE Payments.InvoiceID = ins.InvoiceID

END


GO
